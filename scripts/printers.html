# Define the print server $PrintServer = "put print server name here"
# Get the list of printers $Printers = Get-Printer -ComputerName $PrintServer
# Export the list to a CSV file $Printers | Select-Object Name, ShareName, PortName, DriverName | Export-Csv -Path "put path here" -NoTypeInformation
Write-Host "Printer list exported to path"
